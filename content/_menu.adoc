<script>
    // Toggle submenu de Projetos
    const projetosToggle = document.getElementById('projetos-toggle');
    const projetosMenu = document.getElementById('projetos-menu');
    const projetosArrow = document.getElementById('projetos-arrow');
    
    projetosToggle.addEventListener('click', function() {
        const isOpen = projetosMenu.style.maxHeight && projetosMenu.style.maxHeight !== '0px';
        
        if (isOpen) {
            projetosMenu.style.maxHeight = '0px';
            projetosArrow.style.transform = 'rotate(0deg)';
        } else {
            projetosMenu.style.maxHeight = projetosMenu.scrollHeight + 'px';
            projetosArrow.style.transform = 'rotate(180deg)';
        }
    });
    
    // Formatar data e hora
    function formatarData(date, timezone) {
        const opcoes = {
            year: 'numeric',
            month: '2-digit',
            day: '2-digit',
            hour: '2-digit',
            minute: '2-digit',
            second: '2-digit',
            timeZone: timezone
        };
        return new Intl.DateTimeFormat('pt-BR', opcoes).format(date);
    }
    const agora = new Date();
    document.getElementById('update-gmt').textContent = formatarData(agora, 'UTC');
    document.getElementById('update-br').textContent = formatarData(agora, 'America/Sao_Paulo');
</script>
